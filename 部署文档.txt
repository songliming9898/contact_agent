cd /root
git clone https://github.com/songliming9898/contact_agent.git con_agent
cd con_agent


安装系统依赖
# 安装 MySQL 8.0
yum install -y mysql-server
systemctl start mysqld
systemctl enable mysqld

# 初始化数据库
mysql -u root < init_database.sql
或者 mysql -u root -p
source init_database.sql


#配置环境变量

cp .env.example .env   # 如果有的话，或者直接创建


#安装环境依赖库
pip install -r backend/requirements.txt


#创建数据目录+启动
mkdir -p backend/data/contracts backend/data/json backend/data/chroma

# 导入合同（如果 contracts/ 下有 .docx 文件）
python backend/scripts/import_contracts.py --clear

# 启动后端
uvicorn backend.app.api_server:app --host 0.0.0.0 --port 8000 &
